1.比较C和python两种语言执行效率

Step 1: 求1+2+3+…+n

n = 1 000, 10 000, 100 000, 100 000, 1 000 000, 10 000 000, 100 000 000

Step 2: 输出执行时间；

c语言：

#include<stdio.h>
#include <time.h>
#define n 10000000 
int main()
{
    long long num_sum = 0;
    clock_t start, finish;
    double Total_time;
    start = clock();        //计时开始
    int i;
    for(i=1;i<n+1;i++)
        num_sum=num_sum+i;

    finish = clock();    //计时结束
    Total_time = (double)(finish - start) / CLOCKS_PER_SEC;
    printf("%f seconds\n", Total_time);
    printf("%lf", (double)num_sum);
    return 0;
 }

Python:

print("%f seconds\n" % Total_time)
print("%d"% num_sum)

输出：有序数组

快速排序

基本思想：：参见排序算法专栏

#include<time.h>
#include<stdlib.h>

{
clock_t start, finish;

printf("%d ",arr[i]);

void quickSort(int *arr,int left, int right)//快速排序
{
    if(left >= right)
        return;
    int i, j, base, temp;
    i = left, j = right;
    base = arr[left];  //取最左边的数为基准数
    while (i < j)
    {
        while (arr[j] >= base && i < j)
            j--;
        while (arr[i] <= base && i < j)
            i++;
        if(i < j)
        {
            temp = arr[i];
            arr[i] = arr[j];
            arr[j] = temp;
        }
    }
    //基准数归位
    arr[left] = arr[i];
    arr[i] = base;
    quickSort(arr,left, i - 1 );//递归左边
    quickSort(arr,i + 1, right );//递归右边
}

4.分子量

给出一种物质的分子式(不带括号)，求分子量。本题中的分子式只包含四种原子，分别是C,H,O,N，原子量分别为12.01,1.008,16.00,14.01(单位g/mol)。例如：C6H5OH的分子量为94.108g/mol。

思路：

1.读取输入的字符将其存储到字符串数组中

2.判断本位字符a[i]的后一位字符a[i+1]是否’2’~’9’，如果是则将字符对应 原子量*(a[i+1] -‘0’)

3.重复步骤2

c：

#include<stdio.h>
int main()
{int i=0;
double sum=0;
char c,d[1000];
double lo(char c);
while((c=getchar())!='\n')
{
d[i]=c;
i++;
}
for(;i>=0;i--){
c=d[i];
if(d[i+1]>'2'&&d[i+1]<='9')
{
sum=sum+lo(d[i])*(d[i+1]-'0');
i--;
}
else
sum=lo(d[i])+sum;
}
printf("%.3lfg/mol",sum);
return 0;
}
double lo(char c){
if(c=='C') return 12.01;
if(c=='H') return 1.008;
if(c=='O') return 16.00;
if(c=='N') return 14.01;
}

输入：一个字符串，如：abc

输出：一个逆序后的字符串，如：cba

要求：借助栈这种数据结构来完成

c：

{
 struct Reverse//定义一个结构体
 {
     char c[N];//储存数组
     int topIdx;//顶部指针
 };
 int i=0;
 char ch;
 struct Reverse a, *p;
 a.topIdx=0;
 while((ch=getchar())!='\n')//栈的输入
 {
    if(a.topIdx>=N-1)//判断是否还有空间
    printf("栈已满");
    else
    a.c[a.topIdx++]=ch;//存储字符
 }
 while(a.topIdx>=0

char val = a.c[--a.topIdx];//从顶部向下输出

 }
 return 0;
}
